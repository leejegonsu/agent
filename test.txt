private String getSyncDabangAddress(String address) {
	if (address.startsWith("세종특별자치시")) {
		if (address.startsWith("세종특별자치시 세종") && !address.startsWith("세종특별자치시 세종로")) {
			if (address.startsWith("세종특별자치시 세종시")) {
				return address.replace("세종특별자치시 세종시", "세종특별자치시 세종");
			}
		} else {
			return address.replace("세종특별자치시", "세종특별자치시 세종");
		}
	}
	
	return address;
}
